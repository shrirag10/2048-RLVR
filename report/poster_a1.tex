% ─── Why Custom? ───────────────────────────────────────────────────
\begin{calloutbox}[dqncolor]
{\footnotesize\textbf{\color{dqncolor}Why custom DQN \& SAC?} SB3's DQN lacks direct Q-value masking for invalid actions (only action-space wrappers). Our custom PyTorch implementation masks invalid Q-values to $-\infty$ \emph{inside the forward pass}, enabling proper $\varepsilon$-greedy over legal moves only. SAC required a discrete-action adaptation (Christodoulou, 2019) with categorical policies and batched GPU inference for speed.}
\end{calloutbox}

\sectiontitle{Algorithms}

% ─── DQN ───────────────────────────────────────────────────────────
\begin{sectionbox}[DQN $\bigstar$ Best Performer \hfill \customtag]{dqncolor}

\begin{minipage}[t]{0.48\linewidth}
{\footnotesize
\textbf{Off-policy, value-based}

\renewcommand{\arraystretch}{1.1}
\begin{tabular}{@{}l l@{}}
Buffer:     & 100K transitions \\
Batch:      & 64 \\
$\varepsilon$: & $1.0\!\to\!0.01$ / 100K \\
Target:     & sync every 1K steps \\
LR:         & $10^{-4}$ (Adam) \\
Grad clip:  & 1.0 \\
\end{tabular}
}
\end{minipage}%
\hfill
\begin{minipage}[t]{0.48\linewidth}
\begin{codebox}
{\scriptsize\ttfamily
\textbf{DQN Pseudocode:}\\
1. Sample $(s,a,r,s')$ from $\mathcal{B}$\\
2. Mask invalid $a'$ to $-\infty$\\
3. $y \gets r + \gamma\max_{a'}Q_{\bar{\theta}}(s',a')$\\
4. $\mathcal{L} \gets$ SmoothL1$(Q_\theta(s,a), y)$\\
5. $\theta \gets \theta - \alpha\nabla\mathcal{L}$\\
6. Sync $\bar{\theta} \gets \theta$ every $C$ steps
}
\end{codebox}
\end{minipage}
\end{sectionbox}

\vspace{6pt}

% ─── SAC ──────────────────────────────────────────────────────────
\begin{sectionbox}[SAC (Discrete) \hfill \customtag]{dqncolor}

\begin{minipage}[t]{0.48\linewidth}
{\footnotesize
\textbf{Off-policy, max-entropy}

\renewcommand{\arraystretch}{1.1}
\begin{tabular}{@{}l l@{}}
Buffer:      & 200K transitions \\
Batch:       & 256 \\
Polyak $\tau$: & 0.005 \\
LR:          & $10^{-4}$ (Adam) \\
Warmup:      & 10K steps \\
Target ent.: & $0.4 \cdot \log|\mathcal{A}|$ \\
\end{tabular}
}
\end{minipage}%
\hfill
\begin{minipage}[t]{0.48\linewidth}
\begin{codebox}
{\scriptsize\ttfamily
\textbf{SAC Pseudocode:}\\
1. Sample $(s,a,r,s')$ from $\mathcal{B}$\\
2. Mask invalid actions in $\pi$\\
3. $y \gets r + \gamma(Q_{\min} - \alpha\log\pi)$\\
4. Update twin $Q_1, Q_2$ critics\\
5. Update $\pi$ via entropy objective\\
6. Auto-tune $\alpha$ to target entropy
}
\end{codebox}
\end{minipage}
\end{sectionbox}

% ─── PPO ──────────────────────────────────────────────────────────
\begin{sectionbox}[PPO (MaskablePPO) \hfill \sbtag]{info}

\begin{minipage}[t]{0.48\linewidth}
{\footnotesize
\textbf{On-policy, actor-critic}

\renewcommand{\arraystretch}{1.1}
\begin{tabular}{@{}l l@{}}
Rollout:  & 2048 steps $\times$ 8 envs \\
Batch:    & 64 \\
Epochs:   & 10 \\
LR:       & $3{\times}10^{-4}$ \\
GAE:      & $\lambda{=}0.95$ \\
Clip:     & $\epsilon{=}0.2$ \\
Entropy:  & $c_e{=}0.01$ \\
\end{tabular}
}
\end{minipage}%
\hfill
\begin{minipage}[t]{0.48\linewidth}
\begin{codebox}
{\scriptsize\ttfamily
\textbf{PPO Pseudocode:}\\
1. Collect rollout ($T$ steps, 8 envs)\\
2. Compute GAE advantages $\hat{A}$\\
3. $r_t(\theta) = \frac{\pi_\theta}{\pi_{\text{old}}}$\\
4. $\mathcal{L} = \min(r_t\hat{A}, \text{clip}(r_t)\hat{A})$\\
5. Update $\theta$ for $K{=}10$ epochs
}
\end{codebox}
\end{minipage}
\end{sectionbox}

\vspace{6pt}

% ─── A2C ──────────────────────────────────────────────────────────
\begin{sectionbox}[A2C \hfill \sbtag]{info}
{\footnotesize
\textbf{On-policy, actor-critic.} Simpler than PPO: no clipping, single gradient step per rollout.

\renewcommand{\arraystretch}{1.05}
\begin{tabular}{@{}l l l l@{}}
n\_steps: & 5 & LR: & $7{\times}10^{-4}$ \\
$c_e$: & 0.01 & $c_v$: & 0.5 \\
\end{tabular}

$\mathcal{L} = -\mathbb{E}[\log\pi\hat{A}] + 0.5\|R{-}V\|^2 - 0.01\mathcal{H}[\pi]$
}
\end{sectionbox}

% ─── QR-DQN ──────────────────────────────────────────────────────
\begin{sectionbox}[QR-DQN (Distributional) \hfill \sbtag]{info}
{\footnotesize
\textbf{Off-policy, distributional.} Learns 50 quantiles of return instead of just $\mathbb{E}[Q]$. Net outputs $4{\times}50{=}200$ values; action by mean $Q$.

$\mathcal{L} = \sum_i\rho_{\tau_i}^{\kappa}(\delta_{ij}),\quad \delta_{ij} = r + \gamma\theta_j(s') - \theta_i(s)$

Buffer: 100K \textbar{} $N{=}50$ quantiles \textbar{} LR: $10^{-4}$ \textbar{} Target sync: 1K steps
}
\end{sectionbox}

\vspace{6pt}

% ─── Semi-gradient SARSA ─────────────────────────────────────────
\begin{sectionbox}[Semi-gradient SARSA (S\&B Alg.\ 10.1) \hfill \numpytag]{success}

\begin{minipage}[t]{0.48\linewidth}
{\footnotesize
\textbf{On-policy, linear FA, no neural net!}

$\hat{q}(s,a,\mathbf{w}) = \mathbf{w}_a^\top\phi(s)$

$\mathbf{w}\in\mathbb{R}^{4\times30}$ = \textbf{120 params total}

\renewcommand{\arraystretch}{1.05}
\begin{tabular}{@{}l l@{}}
$\alpha$:   & $10^{-3}$ \\
$\varepsilon$ decay: & 0.99995/step \\
$\varepsilon_{\min}$: & 0.05 \\
\end{tabular}

\textbf{30-dim features} (\S9.5):\\ log-tiles, empty ratio, merge potential, 4 monotonicity, corner, smoothness, snake, edge.
}
\end{minipage}%
\hfill
\begin{minipage}[t]{0.48\linewidth}
\begin{codebox}
{\scriptsize\ttfamily
\textbf{SARSA Pseudocode (Alg 10.1):}\\
1. Observe $s$, choose $a$ ($\varepsilon$-greedy)\\
2. Take action $a$, observe $r, s'$\\
3. Choose $a'$ ($\varepsilon$-greedy from $s'$)\\
4. $\delta \gets r + \gamma\hat{q}(s',a') - \hat{q}(s,a)$\\
5. $\mathbf{w}_a \gets \mathbf{w}_a + \alpha\delta\phi(s)$\\
6. $s \gets s',\; a \gets a'$
}
\end{codebox}
\end{minipage}

\vspace{2pt}
\begin{calloutbox}[success]
{\footnotesize\bfseries 120 params beats 330K+ param deep nets!} Engineered features provide stronger inductive bias than CNNs (S\&B \S9.5).
\end{calloutbox}
\end{sectionbox}

% ─── GRPO ─────────────────────────────────────────────────────────
\begin{sectionbox}[GRPO: Group Relative Policy Optimization \hfill \grpotag]{purple}

{\footnotesize
\textbf{GRPO} = \textbf{G}roup \textbf{R}elative \textbf{P}olicy \textbf{O}ptimization (DeepSeek-R1, 2025).

Fine-tunes \textbf{Qwen2.5-0.5B / 1.5B} LLMs to play 2048 via verifiable rewards, \emph{without a critic network}. Board state $\to$ ASCII text $\to$ model generates reasoning + action.
}

\vspace{4pt}

\begin{minipage}[t]{0.48\linewidth}
{\footnotesize
\textbf{Group-normalized advantage}:
$$\hat{A}_i = \frac{r_i - \mu_G}{\sigma_G + \epsilon}$$
$G{=}4$ completions per prompt.

\textbf{3-stage reward scheduling}:
\begin{enumerate}[nosep, leftmargin=14pt, itemsep=1pt]
    \item Format compliance (XML)
    \item Game reward ($\Delta$score)
    \item Thinking quality
\end{enumerate}
}
\end{minipage}%
\hfill
\begin{minipage}[t]{0.48\linewidth}
\begin{codebox}
{\scriptsize\ttfamily
\textbf{GRPO Pseudocode:}\\
1. Sample $G$ completions per prompt\\
2. Score each with verifier $r_i$\\
3. $\hat{A}_i = (r_i - \mu_G) / \sigma_G$\\
4. $\mathcal{L} = -\sum_i\hat{A}_i\log\pi_\theta$\\
\quad $- \beta\text{KL}(\pi_\theta\|\pi_{\text{ref}})$\\
5. Update $\theta$ via gradient descent
}
\end{codebox}
\end{minipage}
\end{sectionbox}

Interpretable  & No & \color{purple}{\textbf{Yes (CoT)}} \\
\rowcolor{dqnlight}
Observation    & \color{dqncolor}{\textbf{Tensor (16ch)}} & ASCII text \\
Action space   & Numeric & Natural language \\
\bottomrule
\end{tabular}%
}%
}

\vspace{4pt}
{\footnotesize
\textbf{Why classical wins on score}: DQN processes 5M transitions with replay, while the LLM sees only $300{\times}4{=}1200$ game states. DQN's tensor encoding is information-rich; the LLM must parse text.

\textbf{Where RLVR excels}: \emph{Interpretability}. Each LLM move includes a chain-of-thought explanation that can be inspected, debugged, and corrected. Classical agents are black boxes.

\textbf{Model scale matters}: 0.5B $\to$ 1.5B ($3\times$ params) yields $3.3\times$ score improvement, suggesting larger models can better reason about board structure.
}
\end{sectionbox}

% ─── Key Findings ──────────────────────────────────────────────────
\sectiontitle{Key Findings}

\begin{highlightbox}[dqncolor]
{\large\bfseries\color{dqncolor} 1.}\;
{\bfseries Off-policy replay is decisive.} DQN's 100K buffer (S\&B \S8.2) amplifies sparse rewards. On-policy methods fail because they discard data.
\end{highlightbox}

\vspace{4pt}

\begin{highlightbox}[success]
{\large\bfseries\color{success} 2.}\;
{\bfseries Features $>$ deep nets} (at low data). SARSA (120~params) beats all deep on-policy agents (330K+). Validates S\&B~\S9.5.
\end{highlightbox}

\vspace{4pt}

\begin{highlightbox}[danger]
{\large\bfseries\color{danger} 3.}\;
{\bfseries More training hurts PPO.} Score drops 12\% from 1M$\to$5M. QR-DQN develops invalid-move loops.
\end{highlightbox}

\vspace{4pt}

\begin{highlightbox}[purple]
{\large\bfseries\color{purple} 4.}\;
{\bfseries GRPO scales with model size.} $3\times$ more params $\to$ $3.3\times$ higher score. LLMs provide \emph{interpretable} moves.
\end{highlightbox}
